使用Context控制多个goroutine

Nov 19, 2020

Play

package main

import (
	"context"
	"time"
)

type Foo struct {
	ctx    context.Context
	cancel context.CancelFunc
	ch     chan int
}

func (f *Foo) r1() {
	for {
		select {
		case i := <-f.ch:
			println("r1 print: ", i)
		case <-f.ctx.Done():
			println("Foo.r1() routine exit")
			return
		}
	}
}

func (f *Foo) r2() {
	for {
		select {
		case i := <-f.ch:
			println("r2 print: ", i)
		case <-f.ctx.Done():
			println("Foo.r2() routine exit")
			return
		}
	}
}

func (f *Foo) r3() {
	for {
		select {
		case i := <-f.ch:
			println("r3 print: ", i)
		case <-f.ctx.Done():
			println("Foo.r3() routine exit")
			return
		}
	}
}

func main() {
	ctx, cancel := context.WithTimeout(context.Background(), 1*time.Minute)
	f := &Foo{
		ch:     make(chan int),
		ctx:    ctx,
		cancel: cancel,
	}
	go f.r1()
	go f.r2()
	go f.r3()
	for i := 0; i < 100; i++ {
		f.ch <- i
		<-time.After(10 * time.Millisecond)
	}
	go f.cancel()
	<-time.After(1 * time.Second)
	println("main routine exit")
}
